[docs]	def create_study(self, study_id, name, description, homepage, start_date, end_date, researchers=None, attributes=None, *args, **kwargs):
		"""
		Insert a study into the database.

		:param study_id: The study's unique ID.
		:type study_id: str
		:param name: The study's name.
		:type name: str
		:param description: A short description of the study.
		:type description: str
		:param homepage: A link to the study's homepage.
		:type homepage: str
		:param start_date: The date when the study will become active.
		:type start_date: str
		:param end_date: The date when the study finishes.
		:type end_date: str
		:param researchers: A list of researches that are participating in the study.
		:type researchers: list
		:param attributes: A list of attributes that are associated with the study.
		:type attributes: list

		:return: A response with any errors that may arise.
		:rtype: :class:`oauth2.web.Response`
		"""

		response = Response()

		try:
			"""
			Load, parse and sanitize the study arguments.
			"""
			study_id = int(study_id)
			name, description, homepage = self._sanitize_list([ name, description, homepage ])
			start_date = datetime.strptime(start_date, "%Y-%m-%d")
			end_date = datetime.strptime(end_date, "%Y-%m-%d")
			f_start_date = start_date.strftime("%Y-%m-%d")
			f_end_date = end_date.strftime("%Y-%m-%d")

			researchers = [] if researchers is None else researchers
			researchers = self._sanitize_list(researchers)
			attributes = [] if attributes is None else attributes

			"""
			Validate the data.
			"""
			if self._study_exists(study_id):
				raise study_exceptions.StudyExistsException()

			if (end_date - start_date).days <= 0:
				raise study_exceptions.InvalidStudyDurationException()

			for researcher in researchers:
				if not self._researcher_exists(researcher):
					raise user_exceptions.ResearcherDoesNotExistException()

			"""
			Add the study.
			"""
			self._connector.execute([
				"""
				INSERT INTO studies (
					study_id, name, description, homepage, start_date, end_date)
				VALUES (%d, '%s', '%s', '%s', '%s', '%s');""" % (study_id, name, description, homepage, f_start_date, f_end_date),
			])

			"""
			Add the researchers.
			"""
			self._connector.execute([
				"""
				INSERT INTO studies_researchers(
					study_id, researcher_id)
				VALUES (%d, '%s');""" % (study_id, researcher)
			for researcher in researchers ])

			"""
			Add the attributes.
			"""
			for name, type, constraints in attributes:
				if not self._attribute_exists(name, type, constraints):
					self._connector.execute([
						"""
						INSERT INTO attributes(
							name, type, constraints)
						VALUES ('%s', '%s', '%s');""" % (name, type, "{%s}" % ','.join(constraints))
					])
				attribute_id = self._get_attribute_id(name, type, constraints)
				self._connector.execute(
					"""
					INSERT INTO studies_attributes(
						study_id, attribute_id
					)
					VALUES(%d, %d)""" % (study_id, attribute_id)
				)

			response.status_code = 200
			response.add_header("Content-Type", "application/json")
			response.body = json.dumps({ })
		except (general_exceptions.InputException,
				study_exceptions.StudyExistsException,
				user_exceptions.ResearcherDoesNotExistException) as e:
			response.status_code = 500
			response.add_header("Content-Type", "application/json")
			response.body = json.dumps({ "error": str(e), "exception": e.__class__.__name__ })
		except Exception as e:
			response.status_code = 500
			response.add_header("Content-Type", "application/json")
			response.body = json.dumps({ "error": "Internal Server Error: %s" % str(e), "exception": e.__class__.__name__ })

		return response

		"""

		:param study_id: The study's unique ID.
		:type study_id: str
		:param name: The study's name.

		:type start_date: str
		:param end_date: The date when the study finishes.

		:param researchers: A list of researches that are participating in the study.

		:param attributes: A list of attributes that are associated with the study.

				UPDATE studies
				SET
					"name" = '%s',
					"description" = '%s',
					"homepage" = '%s',
					"start_date" = '%s',
					"end_date" = '%s'
				WHERE
					"study_id" = %d;""" % (name, description, homepage, f_start_date, f_end_date, study_id),

[docs]	def get_studies(self, number=10, page=1, search="", case_sensitive=False, *args, **kwargs):
		"""
		Retrieve a list of studies.

		:param number: The number of studies to retrieve.
		:type number: str
		:param page: The page number, used to aid in pagination.
		:type page: str
		:param search: A search string used to look up studies using their name and description.
		:type search: str
		:param case_sensitive: A boolean indicating whether the search should be case sensitive.
		:type case_sensitive: str

		:return: A response containing a list of study objects and any errors that may arise.
		:rtype: :class:`oauth2.web.Response`
		"""

		number = max(int(number), 0)
		page = max(int(page), 1)
		case_sensitive = case_sensitive == "True"

		rows = self._connector.select("""
			SELECT *
			FROM studies
			WHERE
				"name" %s '%%%s%%' OR
				"description" %s '%%%s%%'
			LIMIT %d OFFSET %d
		""" % (
			"LIKE" if case_sensitive else "ILIKE", search,
			"LIKE" if case_sensitive else "ILIKE", search,
			number, number * (page - 1)
		))

		total = self._connector.count("""
			SELECT COUNT(*)
			FROM studies
			WHERE
				"name" %s '%%%s%%' OR
				"description" %s '%%%s%%'
		""" % (
			"LIKE" if case_sensitive else "ILIKE", search,
			"LIKE" if case_sensitive else "ILIKE", search,
		))

		for row in rows:
			row["start_date"] = row["start_date"].isoformat()
			row["end_date"] = row["end_date"].isoformat()

		response = Response()
		response.status_code = 200
		response.add_header("Content-Type", "application/json")
		response.body = json.dumps({ "data": rows, "total": total })
		return response

		"""
		Sanitize all the elements in the given list to be used in PostgreSQL.

		:param l: The list containing all the elements to sanitize.
		:type l: list
		:param sanitize_escaped: A boolean indicating whether escaped characters should be sanitized.
		:type sanitize_escaped: bool

		:return: A new list of sanitized strings.
		:rtype: list
		"""

		return [ self._sanitize(e, sanitize_escaped=sanitize_escaped) for e in l ]



[docs]	def _sanitize_dict(self, d, sanitize_escaped=True):
		"""
		Sanitize all the values in the given dictionary to be used in PostgreSQL.

		:param d: The dict containing all the elements to sanitize.
		:type d: dict
		:param sanitize_escaped: A boolean indicating whether escaped characters should be sanitized.
		:type sanitize_escaped: bool

		:return: A new dict with sanitized values.
		:rtype: dict
		"""

		return { k: self._sanitize(v, sanitize_escaped=sanitize_escaped) for k, v in d.items() }



[docs]	def _sanitize(self, string,
		sanitize_escaped=True):
		"""
		Sanitize the given string to be used in PostgreSQL.

		:param string: The string to sanitize.
		:type string: str
		:param sanitize_escaped: A boolean indicating whether escaped characters should be sanitized.
		:type sanitize_escaped: bool

		:return: The sanitized string.
		:rtype: str
		"""

		"""
		Remove the escape character
		"""
		if sanitize_escaped:
			escape_pattern = "\\\\(.)"
			escape_pattern = re.compile(escape_pattern)
			string = escape_pattern.sub("\g<1>", string)

		"""
		Apostrophes have to be repeated twice.
		"""
		apostrophe_pattern = "'"
		apostrophe_pattern = ("\\\\?" if sanitize_escaped else "") + apostrophe_pattern
		apostrophe_pattern = re.compile(apostrophe_pattern)
		string = apostrophe_pattern.sub("''", string)

		return string
